~~~C
rsum:
	push %rbp
	push %r10
	rrmovq %rsp, %rbp
	push %rbx
	irmovq $24, %r10
	subq %r10, %rsp
	rmmovq %rdi, -24(%rbp)
	rmmovq %rsi, -28(%rbp)
	mrmovq -28(%rbp), %r10
	andq %r10, %r10
	jg .L2
	irmovq $0, %rax
	jmp .L3
.L2:
	mrmovq -24(%rbp), %rax
	mrmovq (%rax), %rbx
	mrmovq -28(%rbp), %rax
	rrmovq %rax, %rdx
	subq $1, %rdx
	mrmovq -24(%rsp), %rax
	irmovq $8, %r10
	addq %r10, %rax
	rrmovq %rdx, %rsi
	rrmovq %rax, %rdi
	call rsum
	addq %rbx, %rax
.L3:
	irmovq $24, %r10
	addq %r10, %rsp
	popq %rbx
	popq %r10
	popq %rbp
	ret
~~~
